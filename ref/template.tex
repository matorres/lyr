\newcommand{\blurred}[4]{
  \transparent{#1}
  \newcount\xend
  \newcount\yend
  \xend=#4
  \yend=#4
  \foreach \xc in {-\xend, ..., \xend}
  \foreach \yc in {-\yend, ..., \yend} {
  \draw[white] node at (#2*\xc/\xend,#2*\yc/\yend) {#3};
  }
}

\newcommand{\blurry}[3]{
\begin{scope}[shift={(0,#2)}]
\newcount\tstep
\def\scale{\fsize/20}
\begin{scope}[on background layer]
	\ifnum#3=1
		\blurred{0.06}{0.02*\scale}{#1}{4}
		\blurred{0.035}{0.05*\scale}{#1}{5}
		\blurred{0.02}{0.10*\scale}{#1}{6}
		\blurred{0.015}{0.15*\scale}{#1}{7}
		\blurred{0.01}{0.20*\scale}{#1}{8}
	\fi
\end{scope}
\transparent{1}%
\uncover<1->{\draw [black] node at (0,0) {#1};}
\end{scope}
}
